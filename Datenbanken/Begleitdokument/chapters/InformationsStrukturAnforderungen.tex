\newcommand{\StartTabelle}[2]{
	\begin{table}[htbp]
		\centering
			\begin{tabular}{llllllll}
		\textbf{Objektbeschreibung:} & #1\\
		\textbf{Anzahl:} & #2\\
		\textbf{Attribute:} \\
		\textbf{Attributname} & \textbf{type} & \textbf{Länge} & \textbf{Wertebereich} & \textbf{\#Wdh.} & \textbf{Def'heit} & \textbf{Id.} \\
}

\newcommand{\DefAttribut}[7]{#1 & #2 & #3 & #4 & #5 & #6 & #7 \\}

\newcommand{\EndeTabelle}{
	\end{tabular}
	\end{table}
}
