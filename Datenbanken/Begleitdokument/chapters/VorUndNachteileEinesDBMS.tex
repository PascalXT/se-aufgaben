\chapter{Vor- und Nachteile Eines DBMS}

\textbf{Vorteile:}
\begin{itemize}
    \item Verhindern von Redundanz und Inkonsistenz
    \item Einheitliche Modellierung der Informationen und damit erweiterte Zugriffsmöglichkeiten
    \item Möglichkeit des Mehrbenutzerbetriebes
    \item Vorbeugung gegen Datenverlust
    \item Durchsetzung von Integritätsbedingungen
    \item Durchsetzung von Sicherheitsmechanismen
    \item Niedrigere Entwicklungskosten für neue Anwendungsprogramme
\end{itemize}



\textbf{Nachteile:}
\begin{itemize}
    \item Kosten für Anschaffung und Betrieb des DBMS
    \item Gegenüber Papierlösungen: Angreifbarkeit durch Computerviren
    \item Mögliche Abhängigkeit von einer einzelnen Software
\end{itemize}
